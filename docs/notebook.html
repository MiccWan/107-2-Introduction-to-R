mergeTags <- function(da, tagNames) {
    # replace all given tags with the first given tag
    resultTag <- tagNames[1]
    
    colsToSummarize <- c(as.character(2016:2017), 'dn')

    da <- da %>%
        # rename all given tags to resultTag
        mutate(tag = ifelse(tag %in% tagNames, resultTag, tag)) %>%
    
        # merge columns with same tag
        group_by(tag) %>%
        summarize_at(colsToSummarize, sum) %>%

        # sort again!
        arrange(desc(dn))

    print(filter(da, tag == resultTag))
    
    print(str_interp('After merging, ${resultTag} has moved from Rank #${match(resultTag, d$tag)} to #${match(resultTag, da$tag)}'))
    
    da
}
